Абстракция

Абстракция (в информатике и программировании) - это обобщение. Уменьшение сложности проблемы.

Пример

def main():
    # Запросить у пользователя длину и ширину прямоугольника.
    length = int(input("Введите длину: "))
    width = int(input("Введите ширину: "))

    # Вызов функции
    rectangle_area = calculate_area(length, width)

    # Напечать результат
    print(f"Площадь треугольника с длиной {length} и шириной {width} равна {rectangle_area}.")


# Создание функции для вычисления площади прямоугольника
def calculate_area(length, width):
    """
    Функция, которая берет два аргумента: длину и ширину прямоугольника и возвращает его площадь.
    """
    area = length * width
    return area


if __name__ == "__main__":
    main()
